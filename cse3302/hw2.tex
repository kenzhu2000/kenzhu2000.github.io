\section*{Problem3 - 40\%}

\begin{enumerate}[(a)]
    \item Give an inductive definition of the judgment form
    $\mbox{leq}\ n_1\ n_2$, which indicates that $n_1$ is less than
    or equal to $n_2$, and $n_1$ and $n_2$ are defined by judgement form 
    $n~ \mbox{nat}$ in the lecture.

    \item Prove by induction: If $n_1~ \mbox{nat}$, $n_2~ \mbox{nat}$, 
	$\mbox{leq}\ n_1\ n_2$
    and $\mbox{leq}\ n_2\ n_1$, then $n_1 = n_2$.
\end{enumerate}
